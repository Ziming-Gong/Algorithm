**Matrix Multiplication**

    A = {   
            a1.1 , a1.2 
            a2.1 , a2.2 
            a3.1 , a3.2 
        }

    B = {   
            b1.1 , b1.2 , b1.3 
            b2.1 , b2.2 , b2.3
        }

    A * B = {
                a1.1*b1.1 + a1.2*b2.1 , a1.1*b1.2 + a1.2*b2.2 , a1.1*b1.3 + a1.2*b2.3
                a2.1*b1.1 + a2.2*b2.1 , a2.1*b1.2 + a2.2*b2.2 , a2.1*b1.3 + a2.2*b2.3
                a3.1*b1.1 + a3.2*b2.1 , a3.1*b1.2 + a3.2*b2.2 , a3.1*b1.3 + a3.2*b2.3

            }

    A (X * N) * B (N * Y) = C (X * Y)
